128/*++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
129 Function: LaTexMathFraction
130
131 Summary: convert decimal_number into fraction form written in string format.
132 Note: it's a modified version of [utilities.h] "DecimalToFraction" fucntion.
133
134 Returns: string
135 store the output directly into string variable.
136------------------------------------------------------------------------------*/
137std::string LaTexMathFraction(const double& decimal_number)
138{
139 int signdec = decimal_number > 0 ? 1 : -1;
140 std::string plusorminus;
141 std::string resulted_string;
142 if(signdec>0) {
143 plusorminus="";
144 }
145 else if(signdec<0) {
146 plusorminus="-";
147 }
148
149 if(!validInteger(abs(decimal_number)) && decimal_number!=0 && decimal_number!=1. && decimal_number!=-1. )
150 {
151
152 double z = decimal_number*decimal_number;
153
154 if(validInteger(z))
155 {
156 //std::cout<<plusorminus+"\sqrt{"<<z;
157 resulted_string =plusorminus+"\\sqrt{"+ToString(z)+"}";
158 }
159
160 else if (!validInteger(z))
161 {
162 int cycles = 10;
163 double precision = 5e-4;
164 double number = z;
165
166 int sign = number > 0 ? 1 : -1;
167 number = number * sign; //abs(number);
168 double new_number,whole_part;
169 double decimal_part = number - (int)number;
170 int counter = 0;
171
172 std::valarray<double> vec_1{double((int) number), 1}, vec_2{1,0}, temporary;
173
174 while( (decimal_part > precision) & (counter < cycles) )
175 {
176 new_number = 1 / decimal_part;
177 whole_part = (int) new_number;
178
179 temporary = vec_1;
180 vec_1 = whole_part * vec_1 + vec_2;
181 vec_2 = temporary;
182
183 decimal_part = new_number - whole_part;
184 counter += 1;
185 }
186 if(validInteger(sqrt(sign * vec_1[0])) && validInteger(sqrt(vec_1[1])))
187 {
188 resulted_string =plusorminus+"\\frac{"+ToString(sqrt(sign * vec_1[0]))+"}{"+ToString(sqrt(vec_1[1]))+"}";
189 }
190
191 else if(validInteger(sqrt(sign * vec_1[0])) && (!validInteger(sqrt(vec_1[1]))) )
192 {
193 resulted_string =plusorminus+"\\frac{"+ToString(sqrt(sign * vec_1[0]))+"}{\\sqrt{"+ToString(vec_1[1])+"}}";
194 }
195 else if( !validInteger(sqrt(sign * vec_1[0])) && (validInteger(sqrt(vec_1[1]))) )
196 {
197 resulted_string =plusorminus+"\\frac{\\sqrt{"+ToString(sign * vec_1[0])+"}}{"+ToString(sqrt(vec_1[1]))+"}";
198 }
199 else
200 resulted_string =plusorminus+"\\sqrt{\\frac{"+ToString(sign * vec_1[0])+"}{"+ToString(sqrt(vec_1[1]))+"}}";
201
202 }
203 }
204
205 else resulted_string=ToString(decimal_number);//std::cout<<decimal_number;
206
207return resulted_string;
208}
209/*++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
210 Function: RealAndImaginary
211
212 Summary: convert tow double variables into string format, with
213 chosen separator string.
214
215 Returns: string
216 store the output directly into string variable.
217-------------------------------------------------------------------*/
218std::string RealAndImaginary(double x, double y, std::string separator)
219{
220 std::string resulted_string;
221 if(x!=0 and y!=0) {
222
223 if(y<0) {
224 if(y==-1){resulted_string = LaTexMathFraction(x)+"-i"+separator;}
225 else resulted_string = LaTexMathFraction(x)+LaTexMathFraction(y)+"i"+separator;
226 }
227
228 else if(y>0) {
229 if(y==1){resulted_string =LaTexMathFraction(x)+"+ i"+separator;}
230 else resulted_string = LaTexMathFraction(x)+"+"+LaTexMathFraction(y)+"i"+separator;
231 }
232 }
233 else if(y==0) {
234 resulted_string = LaTexMathFraction(x)+separator;
235 }
236 else if(x==0){
237 if(y==-1){resulted_string ="-i"+separator;}
238 else if(y==1){resulted_string =" i"+separator;}
239 else resulted_string = LaTexMathFraction(y)+"i"+separator;
240 }
241 return resulted_string;
242}
243/*++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
244 Function: ToLaTex

246 Summary: convert multiple inputs into latex format.

248 Args: const char* fmt...
249 multiple variables with identifier format "fmt".

251 Returns: void
252 print the output directly to the specified latex file

254void LaTex::ToLaTex(const char* fmt...)
255{
256 va_list args;
257 va_start(args, fmt);
258
259 ofile.open(file_path,std::ios_base::app);
260
261 ofile<<"\\begin{align}\n";
262 while (*fmt != '\0') {
263 if (*fmt == 'k') {
264
265 QuantaPlus::Ket<std::complex<double>> mt = va_arg(args, QuantaPlus::Ket<std::complex<double>>);
266
267 ofile<<"\\begin{pmatrix}\n";
268 for(int i =0; i<mt.rows(); i++)
269 {
270 for(int j =0; j<mt.cols(); j++)
271 {
272 double x = std::real(mt(i,j));
273 double y = std::imag(mt(i,j));
274 ofile<<RealAndImaginary(x,y,"\\\\");
275 }
276 }
277 ofile<<"\\end{pmatrix}";
278 }
279 else if (*fmt == 'b') {
280
281 QuantaPlus::Bra<std::complex<double>> mt = va_arg(args, QuantaPlus::Bra<std::complex<double>>);
282
283 ofile<<"\\begin{pmatrix}\n";
284 for(int i =0; i<mt.rows(); i++)
285 {
286 for(int j =0; j<mt.cols(); j++)
287 {
288 double x = std::real(mt(i,j));
289 double y = std::imag(mt(i,j));
290 ofile<<RealAndImaginary(x,y,"\\quad");
291 }
292
293 }
294 ofile<<"\\end{pmatrix}";
295 }
296 else if (*fmt == 'o') {
297
298 QuantaPlus::QM_operator<std::complex<double>> mt = va_arg(args, QuantaPlus::QM_operator<std::complex<double>>);
299
300 ofile<<"\\begin{pmatrix}\n";
301 for(int i =0; i<mt.rows(); i++)
302 {
303 for(int j =0; j<mt.cols(); j++)
304 {
305 double x = std::real(mt(i,j));
306 double y = std::imag(mt(i,j));
307 ofile<<RealAndImaginary(x,y," &");
308 }
309 ofile<<"\\\\";
310 }
311 ofile<<"\\end{pmatrix}";
312 }
313 else if (*fmt == 't') {
314 char* s = va_arg( args, char * );
315 ofile << s;
316 }
317 ++fmt;
318 }
319 va_end(args);
320 ofile<<"\n";
321 ofile<<"\\end{align}\n";
322 ofile.close();
323}
