64void DecimalToFraction(const double& decimal_number )
65{
66
67 int signdec = decimal_number > 0 ? 1 : -1; // check the sign of the input number to select the sign of the final output.
68 std::string plusorminus="";
69
70 if(signdec==-1) {
71 plusorminus="-";
72 }
73 // if the number in not integer and not zero or +1/-1 start the procedure to write it in fraction form
74 if(!IsNumber(ToString(std::abs(decimal_number))) && decimal_number!=0 && decimal_number!=1. && decimal_number!=-1. )
75 {
76
77 double z = decimal_number*decimal_number;
78 // if the square of the number z is an integer, then it can be written as + or - square root of z^2.
79 if(IsNumber(ToString(z)))
80 {
81 std::cout<<plusorminus+"√"<<z;
82 }
83
84 else if (!IsNumber(ToString(z))) // if the square of the number z is not an integer, proceed to write it as fraction.
85 {
86 int cycles = 10;
87 double precision = 5e-4;
88 double number = z;
89
90 int sign = number > 0 ? 1 : -1; // check the sign again but for different reason.
91 number = number * sign; // abs(number);
92 double new_number, whole_part;
93 double decimal_part = number - (int)number; // exclude the integer on the left or the floating point "."
94 int counter = 0;
95
96 std::valarray<double> vec_1{double((int) number), 1}, vec_2{1,0}, temporary;
97
98 while( (decimal_part > precision) & (counter < cycles) )
99 {
100 new_number = 1 / decimal_part;
101 whole_part = (int) new_number;
102
103 temporary = vec_1;
104 vec_1 = whole_part * vec_1 + vec_2;
105 vec_2 = temporary;
106
107 decimal_part = new_number - whole_part;
108 counter += 1;
109 }
110
111 if(IsNumber(ToString(sqrt(sign * vec_1[0]))) && IsNumber(ToString(sqrt(vec_1[1])))) // if both numerator and denominator square roots are integers, evaluate them directly.
112 {
113 std::cout<<plusorminus<<sqrt(sign * vec_1[0])<<'/'<< sqrt(vec_1[1]);
114 }
115
116 // square root of denominator not an integer, write "√denominator" explicitly
117 else if(IsNumber(ToString(sqrt(sign * vec_1[0]))) && (!IsNumber(ToString(sqrt(vec_1[1])))) )
118 {
119 std::cout<<plusorminus<<sqrt(sign * vec_1[0])<<'/'<<"√"<< vec_1[1];
120 }
121 // square root of the numerator is not an integer, write "√numerator" explicitly
122 else if( !IsNumber(ToString(sqrt(sign * vec_1[0]))) && (IsNumber(ToString(sqrt(vec_1[1])))) )
123 {
124 std::cout<<plusorminus+"√"<< sign * vec_1[0]<<'/'<< sqrt(vec_1[1]);
125 }
126 // square root of both numerator and denominator not an integer, write "√(numerator/denominator" explicitly
127 else
128 std::cout<<plusorminus+"√("<< sign * vec_1[0]<<'/'<< vec_1[1]<<")";

130 }
131 }
