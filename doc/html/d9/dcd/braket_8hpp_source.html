83/*!********************************************************************************
84*\brief Function to print QuantaPlus variables (Ket, Bra and C-numbers) in a symbolic form.
85*
86* literal_printf is a "variadic function" takes a variable number of arguments, and
87* like normal printf it needs format specifiers to define the type of data to be printed.
88*@param fmt... is a user defined format specifier.
89*
90\code{.cpp}
91Ket<std::complex<double>> A(3); // deceleration of complex ket vector |A> with three rows.
92Bra<std::complex<double>> B(3); // deceleration of complex Bra vector <B| with three columns.
93
94A <<1.-2.5i, 1i,-0.25i; // initialization of A, |A> = (1-2.5i, i, -0.25i), column vector.
95B << 2.5, 3.7, 2.5+3.5i; // initialization of B, <B| = (2.5, 3.7, 2.5+3.5i), row vector.
96
97literal_printf("kb", A, B); // use format specifier 'kb' to literally print (Ket) A and (Bra) B, respectively.
98\endcode
99*
100\verbatim
101|Ket> =
1021-5/2i
1031i
104-1/4i
105
106<Bra| = 5/2 37/10 5/2+7/2i
107\endverbatim
108***********************************************************************************/
109void literal_printf(const char* fmt...)
110{
111 va_list args; // Type to hold information about variable arguments.
112 va_start(args, fmt); // Initialize a variable argument list.
113
114 while (*fmt != '\0') {
115 if (*fmt == 'k') {
116 Ket<std::complex<double>> kt = va_arg(args, Ket<std::complex<double>>);
117 std::cout<<"|Ket> =\n";
118 ResultPrint(kt);
119 std::cout<<"\n";
120 }
121 else if (*fmt == 'b') {
122 Bra<std::complex<double>> br = va_arg(args, Bra<std::complex<double>>);
123 std::cout<<"<Bra| = ";
124 ResultPrint(br);
125 std::cout<<"\n";
126 }
127 else if (*fmt == 'c') {
128 std::complex<double> d = va_arg(args, std::complex<double>);
129 std::cout<<"complex number = ";
130 ComplexNumPrint(d,0);
131 std::cout<<"\n";
132 }
133 ++fmt;
134 }
135
136 va_end(args); // End using variable argument list.
137}

206Bra<T> DualConj(const Ket<T> &ket)
207{
208 Bra<T> resulted_bra((int)ket.rows());

224/*!****************************************************************
225* \brief Scalar product for Bra and Ket vectors.
226* @return generic type value ( int, double or complex).
227******************************************************************/
228template <typename T>
229T BraKet(const Bra<T> &bra, const Ket<T> &ket)
230{
231 if(bra.cols()!=ket.rows())
232 {
233 throw std::invalid_argument("dimensions do not match.");
234 }
235 return bra*ket;
236}

238/*!***********************************************************************************
239* \brief Compute the expectation value of given operator and normalized quantum state
240**************************************************************************************/
241template <typename T>
242T ExpectValue(const Eigen::Matrix<T,Eigen::Dynamic,Eigen::Dynamic> & mat, const Ket<T> &ket)
243{
244 Bra<T> bra;
245 bra = DualConj(ket);
246 Ket<T> toright((int)ket.rows());
247 toright << mat * ket;
248 T nomr = BraKet(bra,toright);
249 return nomr;//dnomr;
