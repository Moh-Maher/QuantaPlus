168/*++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
169 Function: Hamiltonian
170
171 Summary:
172
173 Args: QM_operator<T> A
174 matrix of type T
175 QM_operator<T> B
176 matrix of type T
177 Bra<T> coff
178 row vector of type T
179
180 Returns: QM_operator<T>
181 Matrix of arbitrary size and generic type T.
182------------------------------------------------------------------*/
183template<typename T>
184QM_operator<T> Hamiltonian(QM_operator<T> A, QM_operator<T> B, Bra<T> coff)
185{
186 int l = (int)coff.cols();
187 QM_operator<T> res((int)A.rows(),(int)A.cols());
188 QM_operator<T> a((int)A.rows(),(int)A.cols());
189 a = -0.25 *B.pow(2);
190
191 QM_operator<T> b((int)A.rows(),(int)A.cols());
192 b = QM_operator<T>::Zero((int)A.rows(),(int)A.cols()) ;
193
194 for(int i=0; i<l; i++){
195 b += coff[i]*pow(0.5,0.5*i)*A.pow(i);
196 }
197
198 res = a + b;
199 res.conservativeResize(res.rows()-1, res.cols()-1);
200 return res;
201}
