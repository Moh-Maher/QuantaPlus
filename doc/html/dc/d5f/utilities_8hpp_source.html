23void DecimalToFraction(const double& decimal_number )
24{
25
26 int signdec = decimal_number > 0 ? 1 : -1; // check the sign of the input number to select the sign of the final output.
27 std::string plusorminus="";
28
29 if(signdec==-1) {
30 plusorminus="-";
31 }
32 // if the number in not integer and not zero or +1/-1 start the procedure to write it in fraction form
33 if(!IsNumber(ToString(std::abs(decimal_number))) && decimal_number!=0 && decimal_number!=1. && decimal_number!=-1. )
34 {
35
36 double z = decimal_number*decimal_number;
37 // if the square of the number z is an integer, then it can be written as + or - square root of z^2.
38 if(IsNumber(ToString(z)))
39 {
40 std::cout<<plusorminus+"√"<<z;
41 }
42
43 else if (!IsNumber(ToString(z))) // if the square of the number z is not an integer, proceed to write it as fraction.
44 {
45 int cycles = 10;
46 double precision = 5e-4;
47 double number = z;
48
49 int sign = number > 0 ? 1 : -1; // check the sign again but for different reason.
50 number = number * sign; // abs(number);
51 double new_number, whole_part;
52 double decimal_part = number - (int)number; // exclude the integer on the left or the floating point "."
53 int counter = 0;
54
55 std::valarray<double> vec_1{double((int) number), 1}, vec_2{1,0}, temporary;
56
57 while( (decimal_part > precision) & (counter < cycles) )
58 {
59 new_number = 1 / decimal_part;
60 whole_part = (int) new_number;
61
62 temporary = vec_1;
63 vec_1 = whole_part * vec_1 + vec_2;
64 vec_2 = temporary;
65
66 decimal_part = new_number - whole_part;
67 counter += 1;
68 }
69
70 if(IsNumber(ToString(sqrt(sign * vec_1[0]))) && IsNumber(ToString(sqrt(vec_1[1])))) // if both numerator and denominator square roots are integers, evaluate them directly.
71 {
72 std::cout<<plusorminus<<sqrt(sign * vec_1[0])<<'/'<< sqrt(vec_1[1]);
73 }
74
75 // square root of denominator not an integer, write "√denominator" explicitly
76 else if(IsNumber(ToString(sqrt(sign * vec_1[0]))) && (!IsNumber(ToString(sqrt(vec_1[1])))) )
77 {
78 std::cout<<plusorminus<<sqrt(sign * vec_1[0])<<'/'<<"√"<< vec_1[1];
79 }
80 // square root of the numerator is not an integer, write "√numerator" explicitly
81 else if( !IsNumber(ToString(sqrt(sign * vec_1[0]))) && (IsNumber(ToString(sqrt(vec_1[1])))) )
82 {
83 std::cout<<plusorminus+"√"<< sign * vec_1[0]<<'/'<< sqrt(vec_1[1]);
84 }
85 // square root of both numerator and denominator not an integer, write "√(numerator/denominator" explicitly
86 else
87 std::cout<<plusorminus+"√("<< sign * vec_1[0]<<'/'<< vec_1[1]<<")";
88
89 }
90 }
91
92 else std::cout<<decimal_number; // if the number in an integer or equal 0, 1 or -1, write it directly as it is.
93}
