14/*!****************************************************************
15* \brief compute specific Eigen vectors for Eigen::Matrix.
16* @param mt is a square Eigen::Matrix with Dynamic-size.
17* @param col_number is the column of the selected eigen vector
18* @return Specified column of eigenvector for a given matrix.
19******************************************************************/
20template<typename T>
21Eigen::Matrix<T,Eigen::Dynamic,1>
