161double ClebschGordan(const double& j1, const double& j2, const double& m1, const double& m2, const double& J, const double& M){
162 if( !validAngularMomentum(j1) || !validAngularMomentum(m1) || !validAngularMomentum(j2) || !validAngularMomentum(m2) || !validAngularMomentum(J) ){
163 throw std::domain_error("clebsch_gordan: all parameters must be multiples of 0.5.");
164 }
165 if( (validInteger(j1) && halfInteger(m1)) || (halfInteger(j1) && validInteger(m1)) ){
166 throw std::domain_error("clebsch_gordan: j1 and m1 must both be integral or half integral.");
167 }
168 if( (validInteger(j2) && halfInteger(m2)) || (halfInteger(j2) && validInteger(m2)) ){
169 throw std::domain_error("clebsch_gordan: j2 and m2 must both be integral or half integral.");
170 }
171 if( (validInteger(j1 + j2) && !validInteger(J)) || (halfInteger(j1 + j2) && !halfInteger(J)) ){
172 throw std::domain_error("clebsch_gordan: J is no valid value for j1 and j2 combination.");
173 }
174 if( j1 < 0 || j2 < 0 ){
175 throw std::domain_error("clebsch_gordan: j1 and j2 must be non-negative.");
176 }
177 if( J > j1 + j2 || J < abs(j1 - j2) ){
178 throw std::domain_error("clebsch_gordan: J must be |j1-j2| <= J <= j1+j2.");
179 }
180 if( abs(m1) > j1 || abs(m2) > j2 || abs(M) > J ){
181 throw std::domain_error("clebsch_gordan: m1 and m2 must be |m1| <= j1 and |m2| <= j2 and |m1+m2| <= J");
182 }
183 if( m1 + m2 != M ){
184 throw std::domain_error("clebsch_gordan: m1 + m2 != M.");
185 }
186
187 //auto const & f = Factorial;
188 double numerator = (2 * J + 1) * Factorial(J + j1 - j2) * Factorial(J - j1 + j2) * Factorial(j1 + j2 - J);
189 numerator *= Factorial(J + M) * Factorial(J - M) * Factorial(j1 - m1) * Factorial(j1 + m1) * Factorial(j2 - m2) * Factorial(j2 + m2);
190 double denominator = Factorial(j1 + j2 + J + 1);
191
192 //int const min = std::max(0., std::max(j2 - J - m1, j1 + m2 - J));
193 //int const max = std::min(j2 + m2, std::min(j1 - m1, j1 + j2 - J));
194 int const min = (int)std::max(0., std::max(j2 - J - m1, j1 + m2 - J));
195 int const max = (int)std::min(j2 + m2, std::min(j1 - m1, j1 + j2 - J));
196 double sum = min > max ? 1 : 0;
197
198 for( int k = min; k <= max; ++k ){
199 sum += pow(-1., k) / (Factorial(k) * Factorial(j1 + j2 - J - k) * Factorial(j1 - m1 - k) * Factorial(j2 + m2 - k) * Factorial(J - j2 + m1 + k) * Factorial(J - j1 - m2 + k));
200 }
201 return SquareRoot(numerator / denominator) * sum;
202}
