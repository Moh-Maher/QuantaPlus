18 Function: SquareRoot
19
20 Summary: gives the given current value if it's equivalent to given previous one, else it
21 will recursively call it self.
22
23 Args: const double& x
24
25 const double& curr
26
27 const double& prev
28
29 Returns: double
30------------------------------------------------------------------*/
31double SquareRoot(const double& x, const double& curr, const double& prev)
32{
33 return curr == prev
34 ? curr
35 : SquareRoot(x, 0.5 * (curr + x / curr), curr);
36}

84 Summary: Calculation of the Clebsch–Gordan Coefficients Generalized power series representations.
85
86 Args:
87
88 Returns:
89------------------------------------------------------------------*/
90double ClebschGordan(const double& j1, const double& j2, const double& m1, const double& m2, const double& J, const double& M){
91 if( !validAngularMomentum(j1) || !validAngularMomentum(m1) || !validAngularMomentum(j2) || !validAngularMomentum(m2) || !validAngularMomentum(J) ){
92 throw std::domain_error("clebsch_gordan: all parameters must be multiples of 0.5.");
93 }
94 if( (validInteger(j1) && halfInteger(m1)) || (halfInteger(j1) && validInteger(m1)) ){
95 throw std::domain_error("clebsch_gordan: j1 and m1 must both be integral or half integral.");
96 }
97 if( (validInteger(j2) && halfInteger(m2)) || (halfInteger(j2) && validInteger(m2)) ){
98 throw std::domain_error("clebsch_gordan: j2 and m2 must both be integral or half integral.");
99 }
100 if( (validInteger(j1 + j2) && !validInteger(J)) || (halfInteger(j1 + j2) && !halfInteger(J)) ){
101 throw std::domain_error("clebsch_gordan: J is no valid value for j1 and j2 combination.");
102 }
103 if( j1 < 0 || j2 < 0 ){
104 throw std::domain_error("clebsch_gordan: j1 and j2 must be non-negative.");
105 }
106 if( J > j1 + j2 || J < abs(j1 - j2) ){
107 throw std::domain_error("clebsch_gordan: J must be |j1-j2| <= J <= j1+j2.");
108 }
109 if( abs(m1) > j1 || abs(m2) > j2 || abs(M) > J ){
110 throw std::domain_error("clebsch_gordan: m1 and m2 must be |m1| <= j1 and |m2| <= j2 and |m1+m2| <= J");
111 }
112 if( m1 + m2 != M ){
113 throw std::domain_error("clebsch_gordan: m1 + m2 != M.");
114 }
115
116 //auto const & f = Factorial;
117 double numerator = (2 * J + 1) * Factorial(J + j1 - j2) * Factorial(J - j1 + j2) * Factorial(j1 + j2 - J);
118 numerator *= Factorial(J + M) * Factorial(J - M) * Factorial(j1 - m1) * Factorial(j1 + m1) * Factorial(j2 - m2) * Factorial(j2 + m2);
119 double denominator = Factorial(j1 + j2 + J + 1);
120
121 int const min = std::max(0., std::max(j2 - J - m1, j1 + m2 - J));
122 int const max = std::min(j2 + m2, std::min(j1 - m1, j1 + j2 - J));
123 double sum = min > max ? 1 : 0;
124
125 for( int k = min; k <= max; ++k ){
126 sum += pow(-1., k) / (Factorial(k) * Factorial(j1 + j2 - J - k) * Factorial(j1 - m1 - k) * Factorial(j2 + m2 - k) * Factorial(J - j2 + m1 + k) * Factorial(J - j1 - m2 + k));
127 }
128 return SquareRoot(numerator / denominator) * sum;
129}

150std::vector<double> MQuantumNumber(const double &j)
151{
152 double spin = j;
153 std::vector<double> m_values;
154 if(validAngularMomentum(spin))

156 for(auto i =-j; i<=j; i++)
157 {
158 if(i==0){m_values.push_back(0);}
159 else m_values.push_back(i);

161 }
162 else{
163 throw std::invalid_argument("the system has physically invalid spin!.");

284 for(auto m = -jj; m <= jj; ++m){
285 for(auto m1 = -j1; m1 <= j1; ++m1) for(auto m2 = -j2; m2 <= j2; ++m2){
286 if( (-jj <= std::abs(m1+m2)) && (std::abs(m1+m2) <= jj) && ( (m1+m2) == m) ){
